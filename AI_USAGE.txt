Here’s a clean implementation of your article controllers with sorting and basic CRUD operations using Node.js and Sequelize:
// Get all articles with optional sorting
exports.getAllArticles = async (req, res) => {
  try {
    let { sortBy, order } = req.query;
    // Default sorting
    if (!sortBy) sortBy = "createdAt";
    if (!order || !["ASC", "DESC"].includes(order.toUpperCase())) order = "DESC";
    const articles = await Article.findAll({
      order: [[sortBy, order.toUpperCase()]],
    });
    res.json(articles);
  } catch (err) {
    res.status(500).json({ error: err.message });
  }
};
